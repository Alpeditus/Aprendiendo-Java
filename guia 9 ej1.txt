package Servicios;

public class CadenaServicio {
    Scanner sc = new Scanner(System.in);
   public void mostrarVocales (String frase ) {
       
      int contador = 0;
        
        for (int i = 0; i < frase.length(); i++) {
  
                 String caracter = frase.substring(i, i+1);
                 
                 if (caracter.equals("a") || caracter.equals("e") || caracter.equals("i") || caracter.equals("o") || caracter.equals("u")) {
                  
                     contador++;
                }
    
        }
        
        System.out.println("Vocales encontradas: " + contador);
        
    }
 //b) Método invertirFrase(), deberá invertir la frase ingresada y mostrarla
//por pantalla. Por ejemplo: Entrada: "casa blanca", Salida: "acnalb
//asac".
public void invertidFrase (String frase) {
    
    StringBuilder sb = new StringBuilder(frase);
    String fraseInvertida = sb.reverse().toString();
    
    System.out.println(fraseInvertida);
    
    
}

frase, por ejemplo:*/
public void vecesRepetido(String letra, String frase){
    
   int contador = 0;
    for (int i = 0; i < frase.length(); i++) {
        if (letra.equals( frase.substring(i, i+1))) {
           contador++; 
        }
        
    }
    System.out.println("La letra aparece " + contador + " veces");
}

public static void reemplazar(String frase, String letra){
        
        String[] retorno = new String[frase.length()];
        
        for (int i = 0; i < frase.length(); i++) {
            if (frase.substring(i, i+1).equals("a")) {
                retorno[i] = letra;
            }else{
                retorno[i] = frase.substring(i, i+1);
            }
        }
        
        String stringResultado = String.join("", retorno);
        System.out.println(stringResultado);
        
        
    }
